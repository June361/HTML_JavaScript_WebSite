1、简单线条

ctx.beginPath(); //开始新路径
ctx.moveTo(10,10); //线条定位起点
ctx.lineTo(20,50); //线条定位终点
ctx.stroke(); //画线条

2、三角形

ctx.beginPath(); //开始新路径
ctx.lineWidth = 10; //线条宽度大小
ctx.lineJoin = "round"; //线条相交时夹角形状 bevel==>创建斜角 round==>创建圆角 miter==>创建尖角(默认)
ctx.moveTo(50,50); //线条定位起点
ctx.lineTo(50,100); //线条定位终点
ctx.lineTo(100,100); //线条定位终点
ctx.closePath(); //创建从当前点到开始点的路径
ctx.stroke(); //画线条

3、填充三角形背景颜色

ctx.beginPath(); //开始新路径
ctx.lineWidth = 10; //线条宽度大小
ctx.lineJoin = "round"; //线条相交时夹角形状 bevel==>创建斜角 round==>创建圆角 miter==>创建尖角(默认)
ctx.moveTo(50,50); //线条定位起点
ctx.lineTo(50,100); //线条定位终点
ctx.lineTo(100,100); //线条定位终点
ctx.closePath(); //创建从当前点到开始点的路径
ctx.stroke(); //画线条
ctx.fillStyle = "color" //填充的颜色设置
ctx.fill(); //进行填充操作

4、线条颜色以及线条大小

ctx.beginPath(); //开始新路径
ctx.lineWidth = 10; //线条宽度大小
ctx.strokeStyle = lineColor; //线条颜色
ctx.moveTo(50,50); //线条定位起点
ctx.lineTo(50,100); //线条定位终点
ctx.stroke(); //画线条

5、二次贝塞尔曲线

ctx.beginPath(); //开始新路径
ctx.moveTo(30,30); //线条定位起点
ctx.quadraticCurveTo(40,100,200,40); //(x1,y1,x2,y2) (x1,y1)为控制点 (x2,y2)为结束点
ctx.stroke(); //画线条

6、三次贝塞尔曲线

ctx.beginPath(); //开始新路径
ctx.moveTo(30,30); //线条定位起点
ctx.bezierCurveTo(30,100,140,100,140,40); //(x1,y1,x2,y2,x3,y3) (x1,y1)为控制点1 (x2,y2)为控制点2 (x3,y3)为结束点
ctx.stroke(); //画线条
